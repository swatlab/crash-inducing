76019: CompileStatus
76019: mjit::Compiler::scanInlineCalls(uint32 index, uint32 depth)
76019: {
76019:     /* Maximum number of calls we will inline at the same site. */
76019:     static const uint32 INLINE_SITE_LIMIT = 5;
76019: 
76019:     JS_ASSERT(inlining());
76019: 
76019:     /* Not inlining yet from 'new' scripts. */
76019:     if (isConstructing)
76019:         return Compile_Okay;
76019: 
76019:     JSScript *script = ssa.getFrame(index).script;
76019:     ScriptAnalysis *analysis = script->analysis(cx);
76019: 
76019:     /* Don't inline from functions which could have a non-global scope object. */
76019:     if (!script->compileAndGo ||
76019:         (script->fun && script->fun->getParent() != globalObj) ||
76019:         (script->fun && script->fun->isHeavyweight()) ||
76019:         script->isActiveEval) {
76019:         return Compile_Okay;
76019:     }
76019: 
76019:     uint32 nextOffset = 0;
76019:     while (nextOffset < script->length) {
76019:         uint32 offset = nextOffset;
76019:         jsbytecode *pc = script->code + offset;
76019:         nextOffset = offset + GetBytecodeLength(pc);
76019: 
76019:         Bytecode *code = analysis->maybeCode(pc);
76019:         if (!code)
76019:             continue;
76019: 
76019:         /* :XXX: Not yet inlining 'new' calls. */
76019:         if (JSOp(*pc) != JSOP_CALL)
76019:             continue;
76019: 
76019:         uint32 argc = GET_ARGC(pc);
76019:         types::TypeSet *calleeTypes = analysis->poppedTypes(pc, argc + 1);
76019: 
76019:         if (calleeTypes->getKnownTypeTag(cx) != JSVAL_TYPE_OBJECT)
76019:             continue;
76019: 
76019:         /*
76019:          * Make sure no callees have had their .arguments accessed, and trigger
76019:          * recompilation if they ever are accessed.
76019:          */
76019:         types::ObjectKind kind = calleeTypes->getKnownObjectKind(cx);
76019:         if (kind != types::OBJECT_INLINEABLE_FUNCTION)
76019:             continue;
76019: 
76019:         if (calleeTypes->getObjectCount() >= INLINE_SITE_LIMIT)
76019:             continue;
76019: 
76019:         /*
76019:          * Compute the maximum height we can grow the stack for inlined frames.
76019:          * We always reserve space for loop temporaries and for an extra stack
76019:          * frame pushed when making a call from the deepest inlined frame.
76019:          */
76019:         uint32 stackLimit = outerScript->nslots + StackSpace::STACK_EXTRA
76019:             - VALUES_PER_STACK_FRAME - FrameState::TEMPORARY_LIMIT;
76019: 
76019:         /* Compute the depth of any frames inlined at this site. */
76019:         uint32 nextDepth = depth + VALUES_PER_STACK_FRAME + script->nfixed + code->stackDepth;
76019: 
76019:         /*
76019:          * Scan each of the possible callees for other conditions precluding
76019:          * inlining. We only inline at a call site if all callees are inlineable.
76019:          */
76019:         unsigned count = calleeTypes->getObjectCount();
76019:         bool okay = true;
76019:         for (unsigned i = 0; i < count; i++) {
76019:             types::TypeObject *object = calleeTypes->getObject(i);
76019:             if (!object)
76019:                 continue;
76019: 
76019:             if (!object->singleton || !object->singleton->isFunction()) {
76019:                 okay = false;
76019:                 break;
76019:             }
76019: 
76019:             JSFunction *fun = object->singleton->getFunctionPrivate();
76019:             if (!fun->isInterpreted()) {
76019:                 okay = false;
76019:                 break;
76019:             }
76019:             JSScript *script = fun->script();
76019: 
76019:             /*
76019:              * The outer and inner scripts must have the same scope. This only
76019:              * allows us to inline calls between non-inner functions. Also
76019:              * check for consistent strictness between the functions.
76019:              */
76019:             if (!script->compileAndGo ||
76019:                 fun->getParent() != globalObj ||
76019:                 outerScript->strictModeCode != script->strictModeCode) {
76019:                 okay = false;
76019:                 break;
76019:             }
76019: 
76019:             /* We can't cope with inlining recursive functions yet. */
76019:             uint32 nindex = index;
76019:             while (nindex != CrossScriptSSA::INVALID_FRAME) {
76019:                 if (ssa.getFrame(nindex).script == script)
76019:                     okay = false;
76019:                 nindex = ssa.getFrame(nindex).parent;
76019:             }
76019:             if (!okay)
76019:                 break;
76019: 
76019:             /* Watch for excessively deep nesting of inlined frames. */
76019:             if (nextDepth + script->nslots >= stackLimit) {
76019:                 okay = false;
76019:                 break;
76019:             }
76019: 
76019:             CompileStatus status = checkAnalysis(script);
76019:             if (status != Compile_Okay)
76019:                 return status;
76019: 
76019:             if (!script->analysis(cx)->inlineable(argc)) {
76019:                 okay = false;
76019:                 break;
76019:             }
76019:         }
76019:         if (!okay)
76019:             continue;
76019: 
76019:         calleeTypes->addFreeze(cx);
76019: 
76019:         /*
76019:          * Add the inline frames to the cross script SSA. We will pick these
76019:          * back up when compiling the call site.
76019:          */
76019:         for (unsigned i = 0; i < count; i++) {
76019:             types::TypeObject *object = calleeTypes->getObject(i);
76019:             if (!object)
76019:                 continue;
76019: 
76019:             JSFunction *fun = object->singleton->getFunctionPrivate();
76019:             JSScript *script = fun->script();
76019: 
76019:             CompileStatus status = addInlineFrame(script, nextDepth, index, pc);
76019:             if (status != Compile_Okay)
76019:                 return status;
76019:         }
76019:     }
76019: 
76019:     return Compile_Okay;
69223: }

75720: CompileStatus
75864: mjit::Compiler::prepareInferenceTypes(JSScript *script, ActiveFrame *a)
75720: {
75987:     /*
75987:      * During our walk of the script, we need to preserve the invariant that at
75987:      * join points the in memory type tag is always in sync with the known type
75987:      * tag of the variable's SSA value at that join point. In particular, SSA
75987:      * values inferred as (int|double) must in fact be doubles, stored either
75987:      * in floating point registers or in memory. (There is an exception for
75987:      * locals with a dead value at the current point, whose type may or may not
75987:      * be synced).
75987:      *
75987:      * To ensure this, we need to know the SSA values for each variable at each
75987:      * join point, which the SSA analysis does not store explicitly. These can
75987:      * be recovered, though. During the forward walk, the SSA value of a var
75987:      * (and its associated type set) change only when we see an explicit assign
75987:      * to the var or get to a join point with a phi node for that var. So we
75987:      * can duplicate the effects of that walk here by watching for writes to
75987:      * vars (updateVarTypes) and new phi nodes at join points.
75987:      *
75987:      * When we get to a branch and need to know a variable's value at the
75987:      * branch target, we know it will either be a phi node at the target or
75987:      * the variable's current value, as no phi node is created at the target
75987:      * only if a variable has the same value on all incoming edges.
75987:      */
75987: 
75987:     a->varTypes = (VarType *)
76019:         cx->calloc_(TotalSlots(script) * sizeof(VarType));
75987:     if (!a->varTypes)
75708:         return Compile_Error;
75987:

75630: void
75987: mjit::Compiler::fixDoubleTypes(jsbytecode *target)
75630: {
75720:     if (!cx->typeInferenceEnabled())
75720:         return;
75720: 
75987:     /*
75998:      * Fill fixedDoubleEntries with all variables that are known to be an int
75998:      * here and a double at the branch target. Per prepareInferenceTypes, the
75998:      * target state consists of the current state plus any phi nodes or other
75998:      * new values introduced at the target.
75987:      */
76028:     JS_ASSERT(fixedDoubleEntries.empty());
76019:     const SlotValue *newv = analysis->newValues(target);
75987:     if (newv) {
75987:         while (newv->slot) {
76019:             if (newv->value.kind() != SSAValue::PHI ||
75987:                 newv->value.phiOffset() != uint32(target - script->code)) {
75987:                 newv++;
75987:                 continue;
75987:             }
76019:             if (newv->slot < TotalSlots(script)) {
76008:                 types::TypeSet *targetTypes = analysis->getValueTypes(newv->value);
75987:                 VarType &vt = a->varTypes[newv->slot];
75998:                 if (targetTypes->getKnownTypeTag(cx) == JSVAL_TYPE_DOUBLE &&
75998:                     fixDoubleSlot(newv->slot)) {
76019:                     FrameEntry *fe = frame.getSlotEntry(newv->slot);
76008:                     if (vt.type == JSVAL_TYPE_INT32) {
75998:                         fixedDoubleEntries.append(newv->slot);
75998:                         frame.ensureDouble(fe);
76008:                     } else if (vt.type == JSVAL_TYPE_UNKNOWN) {
76008:                         /*
76008:                          * Unknown here but a double at the target. The type
76008:                          * set for the existing value must be empty, so this
76008:                          * code is doomed and we can just mark the value as
76008:                          * a double.
76008:                          */
76008:                         frame.ensureDouble(fe);
76008:                     } else {
76008:                         JS_ASSERT(vt.type == JSVAL_TYPE_DOUBLE);
75998:                     }
75998:                 }
75987:             }
75987:             newv++;
75987:         }
75987:     }
57671: }

75987: mjit::Compiler::restoreAnalysisTypes()
75622: {
75720:     if (!cx->typeInferenceEnabled())
75720:         return;
75720: 
75987:     /* Update variable types for all new values at this bytecode. */

75987: {
75987:     if (!cx->typeInferenceEnabled())
75987:         return;
75987: 
75987:     /*
75987:      * For any non-escaping variable written at the current opcode, update the
75987:      * associated type sets according to the written type, keeping the type set
75987:      * for each variable in sync with what the SSA analysis has determined
75987:      * (see prepareInferenceTypes).
75987:      */

75987:     types::TypeSet *types = NULL;
75987:     switch (JSOp(*PC)) {
75987:       case JSOP_SETARG:
75987:       case JSOP_SETLOCAL:
75987:       case JSOP_SETLOCALPOP:
75987:       case JSOP_DEFLOCALFUN:
75987:       case JSOP_DEFLOCALFUN_FC:

75987:       case JSOP_DECARG:
75987:       case JSOP_ARGINC:
75987:       case JSOP_ARGDEC:
75987:       case JSOP_INCLOCAL:
75987:       case JSOP_DECLOCAL:
75987:       case JSOP_LOCALINC:
75987:       case JSOP_LOCALDEC:
75987:         types = pushedTypeSet(0);
75987:         break;
75987:       case JSOP_FORARG:
75987:       case JSOP_FORLOCAL:
75987:         types = pushedTypeSet(1);
75987:         break;
75987:       default:
75987:         JS_NOT_REACHED("Bad op");
75987:     }
75987:

75987: 
75987:     if (analysis->trackSlot(slot)) {
75987:         VarType &vt = a->varTypes[slot];

75987:         vt.type = types->getKnownTypeTag(cx);
75987:     }
75987: }
