    1: /**
    1:  * The nsIDOMWindow interface is the primary interface for a DOM
    1:  * window object. It represents a single window object that may
    1:  * contain child windows if the document in the window contains a
    1:  * HTML frameset document or if the document contains iframe elements.
    1:  *
68870:  * This interface is not officially defined by any standard bodies, it
68870:  * originates from the defacto DOM Level 0 standard.
    1:  */
    1: 
68870: [scriptable, uuid(a6cf906b-15b3-11d2-932e-00805f8add32)]
    1: interface nsIDOMWindow : nsISupports
    1: {
    1:   /**
    1:    * Accessor for the document in this window.
    1:    */
    1:   readonly attribute nsIDOMDocument                     document;
    1: 
    1:   /**
    1:    * Accessor for this window's parent window, or the window itself if
    1:    * there is no parent, or if the parent is of different type
    1:    * (i.e. this does not cross chrome-content boundaries).
    1:    */
    1:   readonly attribute nsIDOMWindow                       parent;
    1: 
    1:   /**
    1:    * Accessor for the root of this hierarchy of windows. This root may
    1:    * be the window itself if there is no parent, or if the parent is
    1:    * of different type (i.e. this does not cross chrome-content
    1:    * boundaries).
    1:    *
    1:    * This property is "replaceable" in JavaScript */
    1:   readonly attribute nsIDOMWindow                       top;
    1: 
    1:   /**
    1:    * Accessor for the object that controls whether or not scrollbars
    1:    * are shown in this window.
    1:    *
    1:    * This attribute is "replaceable" in JavaScript
    1:    */
    1:   readonly attribute nsIDOMBarProp                      scrollbars;
    1: 
    1:   /**
    1:    * Accessor for the child windows in this window.
    1:    */
    1:   [noscript] readonly attribute nsIDOMWindowCollection  frames;
    1: 
    1:   /**
    1:    * Set/Get the name of this window.
    1:    *
    1:    * This attribute is "replaceable" in JavaScript
    1:    */
    1:            attribute DOMString                          name;
    1:   /**
    1:    * Set/Get the document scale factor as a multiplier on the default
    1:    * size. When setting this attribute, a NS_ERROR_NOT_IMPLEMENTED
    1:    * error may be returned by implementations not supporting
    1:    * zoom. Implementations not supporting zoom should return 1.0 all
    1:    * the time for the Get operation. 1.0 is equals normal size,
    1:    * i.e. no zoom.
    1:    */
    1:   [noscript] attribute float                            textZoom;
    1: 
    1:   /**
    1:    * Accessor for the current x scroll position in this window in
    1:    * pixels.
    1:    *
    1:    * This attribute is "replaceable" in JavaScript
    1:    */
    1:   readonly attribute long                               scrollX;
    1: 
    1:   /**
    1:    * Accessor for the current y scroll position in this window in
    1:    * pixels.
    1:    *
    1:    * This attribute is "replaceable" in JavaScript
    1:    */
    1:   readonly attribute long                               scrollY;
    1: 
    1:   /**
    1:    * Method for scrolling this window to an absolute pixel offset.
    1:    */
    1:   void                      scrollTo(in long xScroll, in long yScroll);
    1: 
    1:   /**
    1:    * Method for scrolling this window to a pixel offset relative to
    1:    * the current scroll position.
    1:    */
    1:   void                      scrollBy(in long xScrollDif, in long yScrollDif);
    1: 
    1:   /**
    1:    * Method for accessing this window's selection object.
    1:    */
    1:   nsISelection              getSelection();
    1: 
    1:   /**
    1:    * Method for scrolling this window by a number of lines.
    1:    */
    1:   void                      scrollByLines(in long numLines);
    1: 
    1:   /**
    1:    * Method for scrolling this window by a number of pages.
    1:    */
    1:   void                      scrollByPages(in long numPages);
    1: 
    1:   /**
    1:    * Method for sizing this window to the content in the window.
    1:    */
    1:   void                      sizeToContent();
    1: };
