  3869: // file_id.h: Return a unique identifier for a file
  3869: //
  3869: // Author: Dan Waylonis
  3869: 
  3869: #ifndef COMMON_MAC_FILE_ID_H__
  3869: #define COMMON_MAC_FILE_ID_H__
  3869: 
  3869: #include <limits.h>
118328: #include <mach/machine.h>
  3869: 
  3869: namespace google_breakpad {
  3869: 
  3869: class FileID {
  3869:  public:
  3869:   FileID(const char *path);
  3869:   ~FileID() {};
  3869: 
  3869:   // Load the identifier for the file path specified in the constructor into
  3869:   // |identifier|.  Return false if the identifier could not be created for the
  3869:   // file.
  3869:   // The current implementation will return the MD5 hash of the file's bytes.
  3869:   bool FileIdentifier(unsigned char identifier[16]);
  3869: 
  3869:   // Treat the file as a mach-o file that will contain one or more archicture.
118328:   // Accepted values for |cpu_type| and |cpu_subtype| (e.g., CPU_TYPE_X86 or
118328:   // CPU_TYPE_POWERPC) are listed in /usr/include/mach/machine.h.
118328:   // If |cpu_type| is 0, then the native cpu type is used. If |cpu_subtype| is
118328:   // CPU_SUBTYPE_MULTIPLE, the match is only done on |cpu_type|.
118328:   // Returns false if opening the file failed or if the |cpu_type|/|cpu_subtype|
118328:   // is not present in the file.
  3869:   // Return the unique identifier in |identifier|.
  3869:   // The current implementation will look for the (in order of priority):
  3869:   // LC_UUID, LC_ID_DYLIB, or MD5 hash of the given |cpu_type|.
118328:   bool MachoIdentifier(cpu_type_t cpu_type,
118328:                        cpu_subtype_t cpu_subtype,
118328:                        unsigned char identifier[16]);
  3869: 
  3869:   // Convert the |identifier| data to a NULL terminated string.  The string will
  3869:   // be formatted as a UUID (e.g., 22F065BB-FC9C-49F7-80FE-26A7CEBD7BCE).
  3869:   // The |buffer| should be at least 37 bytes long to receive all of the data
  3869:   // and termination.  Shorter buffers will contain truncated data.
  3869:   static void ConvertIdentifierToString(const unsigned char identifier[16],
  3869:                                         char *buffer, int buffer_length);
  3869: 
  3869:  private:
  3869:   // Storage for the path specified
  3869:   char path_[PATH_MAX];
  3869: };
  3869: 
  3869: }  // namespace google_breakpad
  3869: 
  3869: #endif  // COMMON_MAC_FILE_ID_H__
