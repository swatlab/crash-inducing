    1: #include "nsIURL.idl"
    1: 
    1: interface nsIFile;
    1: 
    1: /**
    1:  * nsIFileURL provides access to the underlying nsIFile object corresponding to
    1:  * an URL.  The URL scheme need not be file:, since other local protocols may
    1:  * map URLs to files (e.g., resource:).
    1:  */
74388: [scriptable, uuid(44c14c16-07b4-48fb-b44b-0c8986697a17)]
    1: interface nsIFileURL : nsIURL
    1: {
    1:     /**
    1:      * Get/Set nsIFile corresponding to this URL.
    1:      *
    1:      *  - Getter returns a reference to an immutable object.  Callers must clone
    1:      *    before attempting to modify the returned nsIFile object.  NOTE: this
    1:      *    constraint might not be enforced at runtime, so beware!!
    1:      *
    1:      *  - Setter clones the nsIFile object (allowing the caller to safely modify
    1:      *    the nsIFile object after setting it on this interface).
    1:      */
    1:     attribute nsIFile file;
    1: };
