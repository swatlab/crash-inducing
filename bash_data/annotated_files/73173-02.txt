42290: #include "nsIURL.idl"
    1: 
    1:   /**
    1:    * nsIIconURI
    1:    *
    1:    * This interface derives from nsIURI, to provide additional information
    1:    * about moz-icon URIs.
    1:    *
    1:    * What *is* a moz-icon URI you ask?  Well, it has the following syntax:
    1:    *
42290:    * moz-icon:[<valid-url> | //<file-with-extension> | //stock/<stock-icon>]? ['?'[<parameter-value-pairs>]]
    1:    *
42290:    * <valid-url> is a valid URL spec.
    1:    *
    1:    * <file-with-extension> is any filename with an extension, e.g. "dummy.html".
42290:    * If the file you want an icon for isn't known to exist, you can use this instead of a URL and just
42290:    * place a dummy file name with the extension or content type you want.
    1:    *
42290:    * <stock-icon> is the name of a platform-dependant stock icon.
    1:    *
    1:    * Legal parameter value pairs are listed below:
    1:    *
    1:    *   Parameter:   size
    1:    *   Values:      [<integer> | button | toolbar | toolbarsmall | menu | dialog]
    1:    *   Description: If integer, this is the desired size in square pixels of the icon
    1:    *                Else, use the OS default for the specified keyword context.
    1:    *
    1:    *   Parameter:   state
    1:    *   Values:      [normal | disabled]
    1:    *   Description: The state of the icon.
    1:    *
    1:    *   Parameter:   contentType
    1:    *   Values:      <mime-type>
    1:    *   Description: The mime type we want an icon for. This is ignored by stock images.
    1:    */
    1:
