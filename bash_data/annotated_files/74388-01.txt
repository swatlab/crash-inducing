    1: /**
    1:  * JAR URLs have the following syntax
    1:  *
    1:  * jar:<jar-file-uri>!/<jar-entry>
    1:  *
    1:  * EXAMPLE: jar:http://www.big.com/blue.jar!/ocean.html
    1:  *
    1:  * The nsIURL methods operate on the <jar-entry> part of the spec.
    1:  */
74388: [scriptable, uuid(c95d481a-c0ec-43cc-8320-43842b1df597)]
    1: interface nsIJARURI : nsIURL {
    1: 
    1:     /**
    1:      * Returns the root URI (the one for the actual JAR file) for this JAR
    1:      * (e.g., http://www.big.com/blue.jar).
    1:      */
    1:     readonly attribute nsIURI JARFile;
    1: 
    1:     /**
    1:      * Returns the entry specified for this JAR URI (e.g., "ocean.html").  This
    1:      * value may contain %-escaped byte sequences.
    1:      */
    1:     attribute AUTF8String JAREntry;
 8361: 
 8361:     /**
 8361:      * Create a clone of the JAR URI with a new root URI (the URI for the
 8361:      * actual JAR file).
 8361:      */
 8361:     nsIJARURI cloneWithJARFile(in nsIURI jarFile);
    1: };
